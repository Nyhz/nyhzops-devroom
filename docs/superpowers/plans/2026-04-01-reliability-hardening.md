# Reliability Hardening Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Decouple review from merge (Overseer + Quartermaster), harden real-time updates via centralized socket emitter, eliminate data corruption with transactions, and surface errors through inline UI panels.

**Architecture:** Split Captain into Overseer (review-only) and Quartermaster (merge-only). Add centralized `emitStatusChange()` for all socket emissions. Wrap all multi-step DB writes in transactions. Add `approved` and `merging` statuses to the mission lifecycle. Surface all failures as inline error panels with contextual actions.

**Tech Stack:** Next.js 16 App Router, TypeScript strict, Drizzle ORM + better-sqlite3, Socket.IO, Vitest + Testing Library, Playwright

```typescript
import { eq } from 'drizzle-orm';
import { revalidatePath } from 'next/cache';
import { getDatabase } from '@/lib/db/index';
import { missions, phases, campaigns, battlefields } from '@/lib/db/schema';

type Entity = 'mission' | 'phase' | 'campaign' | 'battlefield';

interface ResolvedContext {
  battlefieldId?: string;
  campaignId?: string;
  phaseId?: string;
}

function resolveContext(entity: Entity, id: string): ResolvedContext {
  const db = getDatabase();

  switch (entity) {
    case 'mission': {
      const row = db.select({
        battlefieldId: missions.battlefieldId,
        campaignId: missions.campaignId,
        phaseId: missions.phaseId,
      }).from(missions).where(eq(missions.id, id)).get();
      return {
        battlefieldId: row?.battlefieldId,
        campaignId: row?.campaignId ?? undefined,
        phaseId: row?.phaseId ?? undefined,
      };
    }
    case 'phase': {
      const row = db.select({
        campaignId: phases.campaignId,
      }).from(phases).where(eq(phases.id, id)).get();
      const campaign = row?.campaignId
        ? db.select({ battlefieldId: campaigns.battlefieldId })
            .from(campaigns).where(eq(campaigns.id, row.campaignId)).get()
        : undefined;
      return {
        campaignId: row?.campaignId,
        battlefieldId: campaign?.battlefieldId,
      };
    }
    case 'campaign': {
      const row = db.select({
        battlefieldId: campaigns.battlefieldId,
      }).from(campaigns).where(eq(campaigns.id, id)).get();
      return { battlefieldId: row?.battlefieldId };
    }
    case 'battlefield':
      return {};
  }
}

function revalidateEntity(entity: Entity, id: string, ctx: ResolvedContext): void {
  try {
    if (ctx.battlefieldId) {
      revalidatePath(`/battlefields/${ctx.battlefieldId}`);
    }
    if (ctx.campaignId) {
      revalidatePath(`/battlefields/${ctx.battlefieldId}/campaigns/${ctx.campaignId}`);
    }
    if (entity === 'battlefield') {
      revalidatePath(`/battlefields/${id}`);
      revalidatePath('/');
    }
  } catch {
    // revalidatePath may throw outside of request context — safe to ignore
  }
}

export function emitStatusChange(
  entity: Entity,
  id: string,
  status: string,
  extra?: Record<string, unknown>,
): void {
  const ctx = resolveContext(entity, id);

  // Revalidate cache BEFORE emitting events
  revalidateEntity(entity, id, ctx);

  const io = globalThis.io;
  if (!io) return;

  const timestamp = Date.now();

  switch (entity) {
    case 'mission': {
      const payload = { missionId: id, status, timestamp, ...extra };
      io.to(`mission:${id}`).emit('mission:status', payload);
      if (ctx.battlefieldId) {
        io.to(`battlefield:${ctx.battlefieldId}`).emit('mission:status', payload);
      }
      if (ctx.campaignId) {
        io.to(`campaign:${ctx.campaignId}`).emit('mission:status', payload);
      }
      io.to('hq:activity').emit('mission:status', payload);
      break;
    }
    case 'phase': {
      const payload = { phaseId: id, status, timestamp, ...extra };
      if (ctx.campaignId) {
        io.to(`campaign:${ctx.campaignId}`).emit('phase:status', payload);
      }
      if (ctx.battlefieldId) {
        io.to(`battlefield:${ctx.battlefieldId}`).emit('phase:status', payload);
      }
      io.to('hq:activity').emit('phase:status', payload);
      break;
    }
    case 'campaign': {
      const payload = { campaignId: id, status, timestamp, ...extra };
      io.to(`campaign:${id}`).emit('campaign:status', payload);
      if (ctx.battlefieldId) {
        io.to(`battlefield:${ctx.battlefieldId}`).emit('campaign:status', payload);
      }
      io.to('hq:activity').emit('campaign:status', payload);
      break;
    }
    case 'battlefield': {
      const payload = { battlefieldId: id, status, timestamp, ...extra };
      io.to(`battlefield:${id}`).emit('battlefield:status', payload);
      io.to('hq:activity').emit('battlefield:status', payload);
      break;
    }
  }
}
```

```typescript
import { escalate } from '@/lib/overseer/escalation';

/**
 * Safely trigger orchestrator to pick up a queued mission.
 * Wraps the call in try-catch — on failure, logs and escalates
 * instead of letting the error propagate and leave the mission stuck.
 */
export function safeQueueMission(missionId: string): void {
  try {
    globalThis.orchestrator?.onMissionQueued(missionId);
  } catch (err) {
    console.error(`[SafeQueue] Failed to queue mission ${missionId}:`, err);
    escalate({
      level: 'critical',
      title: 'Mission queue failed',
      detail: `Mission ${missionId} could not be queued: ${err instanceof Error ? err.message : String(err)}. The mission is in QUEUED status but the orchestrator did not pick it up.`,
      entityType: 'mission',
      entityId: missionId,
      battlefieldId: '',
    }).catch(() => { /* best effort */ });
  }
}
```

```typescript
import type { OverseerReview } from '@/types';

type ParseResult =
  | { ok: true; review: OverseerReview }
  | { ok: false; fallback: OverseerReview; diagnostic: string };

const VALID_VERDICTS = ['approve', 'retry', 'escalate'] as const;
type Verdict = typeof VALID_VERDICTS[number];

/** Map legacy field values to current verdict values */
const VERDICT_ALIASES: Record<string, Verdict> = {
  accept: 'approve',
  approve: 'approve',
  retry: 'retry',
  escalate: 'escalate',
};

function escalateFallback(diagnostic: string): ParseResult {
  return {
    ok: false,
    fallback: { verdict: 'escalate', concerns: [], reasoning: diagnostic },
    diagnostic,
  };
}

function coerceConcerns(raw: unknown): string[] {
  if (Array.isArray(raw)) return raw.filter((c): c is string => typeof c === 'string');
  if (typeof raw === 'string') return [raw];
  return [];
}

function extractVerdictFromObj(obj: Record<string, unknown>): Verdict | null {
  // Try verdict field first, then legacy recommendation field
  const raw = obj.verdict ?? obj.recommendation;
  if (typeof raw !== 'string') return null;
  return VERDICT_ALIASES[raw.toLowerCase()] ?? null;
}

function tryParseStructuredOutput(parsed: Record<string, unknown>): ParseResult {
  // Envelope format: { subtype, structured_output, ... }
  if (parsed.subtype !== undefined) {
    if (parsed.subtype !== 'success' || !parsed.structured_output) {
      return escalateFallback(
        `Envelope subtype="${parsed.subtype}", is_error=${parsed.is_error}, result="${String(parsed.result ?? '').slice(0, 200)}"`,
      );
    }
    const so = parsed.structured_output as Record<string, unknown>;
    return tryExtractReview(so);
  }

  // Direct JSON object (no envelope)
  return tryExtractReview(parsed);
}

function tryExtractReview(obj: Record<string, unknown>): ParseResult {
  const verdict = extractVerdictFromObj(obj);
  if (!verdict) {
    return escalateFallback(`Invalid or missing verdict: "${obj.verdict ?? obj.recommendation}"`);
  }

  return {
    ok: true,
    review: {
      verdict,
      concerns: coerceConcerns(obj.concerns),
      reasoning: typeof obj.reasoning === 'string' && obj.reasoning.length > 0
        ? obj.reasoning
        : 'No reasoning provided',
    },
  };
}

function tryExtractJsonFromProse(raw: string): Record<string, unknown> | null {
  // Try code block first
  const codeBlockMatch = raw.match(/```(?:json)?\s*\n?([\s\S]*?)\n?```/);
  if (codeBlockMatch) {
    try { return JSON.parse(codeBlockMatch[1].trim()); } catch { /* continue */ }
  }

  // Try finding a JSON object in the text
  const braceMatch = raw.match(/\{[\s\S]*\}/);
  if (braceMatch) {
    try { return JSON.parse(braceMatch[0]); } catch { /* continue */ }
  }

  return null;
}

export function parseReviewOutput(raw: string): ParseResult {
  // Attempt 1: Direct JSON parse
  try {
    const parsed = JSON.parse(raw);
    if (typeof parsed === 'object' && parsed !== null) {
      return tryParseStructuredOutput(parsed);
    }
  } catch { /* not valid JSON — try extraction */ }

  // Attempt 2: Extract JSON from prose
  const extracted = tryExtractJsonFromProse(raw);
  if (extracted) {
    return tryExtractReview(extracted);
  }

  // Complete failure
  return escalateFallback(
    `Could not parse review output. Raw (first 300 chars): ${raw.slice(0, 300)}`,
  );
}
```

```typescript
import { spawn } from 'child_process';
import fs from 'fs';
import simpleGit from 'simple-git';
import { config } from '@/lib/config';
import { createAuthenticatedHome } from '@/lib/process/claude-print';
import type { Mission } from '@/types';

const RESOLUTION_TIMEOUT_MS = 10 * 60 * 1000; // 10 minutes

export async function resolveConflicts(params: {
  repoPath: string;
  sourceBranch: string;
  targetBranch: string;
  mission: Mission;
  claudeMdPath?: string | null;
}): Promise<boolean> {
  const { repoPath, sourceBranch, targetBranch, mission, claudeMdPath } = params;
  const git = simpleGit(repoPath);

  // Gather rich context
  let conflictDiff = '';
  try { conflictDiff = await git.diff(); } catch { conflictDiff = 'Unable to retrieve conflict diff.'; }

  let upstreamLog = '';
  try { upstreamLog = await git.log({ from: sourceBranch, to: targetBranch, format: { hash: '%h', message: '%s' } }).then(l => l.all.map(c => `${c.hash} ${c.message}`).join('\n')); } catch { /* empty */ }

  let branchLog = '';
  try { branchLog = await git.log({ from: targetBranch, to: sourceBranch, format: { hash: '%h', message: '%s' } }).then(l => l.all.map(c => `${c.hash} ${c.message}`).join('\n')); } catch { /* empty */ }

  const sections: string[] = [];

  if (claudeMdPath) {
    try { sections.push(fs.readFileSync(claudeMdPath, 'utf-8')); } catch { /* skip */ }
  }

  sections.push([
    '## Merge Conflict Resolution',
    '',
    `Branch \`${sourceBranch}\` into \`${targetBranch}\`.`,
    '',
    '### Mission Briefing',
    mission.briefing,
    '',
    '### Mission Debrief',
    mission.debrief || 'No debrief available.',
    '',
    '### What this mission changed',
    '```', branchLog || 'No commits found.', '```',
    '',
    '### What landed upstream since branch diverged',
    '```', upstreamLog || 'No upstream changes found.', '```',
    '',
    '### Conflicts',
    '```', conflictDiff, '```',
    '',
    '### Orders',
    '1. Analyze both sides of each conflict.',
    '2. Resolve preserving both intents.',
    '3. If incompatible, prefer source (new work). Note losses.',
    '4. Run tests if a test command is available.',
    `5. Commit: "Merge ${sourceBranch}: resolve conflicts"`,
    '6. Report to the Commander.',
  ].join('\n'));

  const prompt = sections.join('\n\n---\n\n');
  const authHome = createAuthenticatedHome();

  return new Promise<boolean>((resolve) => {
    const proc = spawn(config.claudePath, [
      '--print',
      '--dangerously-skip-permissions',
      '--max-turns', '20',
    ], {
      cwd: repoPath,
      env: { ...process.env, HOME: authHome },
    });

    const timeout = setTimeout(() => {
      proc.kill('SIGTERM');
      resolve(false);
    }, RESOLUTION_TIMEOUT_MS);

    proc.stdout?.resume();
    proc.stdin?.write(prompt);
    proc.stdin?.end();

    proc.on('close', (code) => {
      clearTimeout(timeout);
      try { fs.rmSync(authHome, { recursive: true, force: true }); } catch { /* best effort */ }
      resolve(code === 0);
    });

    proc.on('error', () => {
      clearTimeout(timeout);
      resolve(false);
    });
  });
}
```

```typescript
import fs from 'fs';
import simpleGit from 'simple-git';
import { resolveConflicts } from './conflict-resolver';
import type { Mission, MergeResult } from '@/types';

const RETRY_DELAY_MS = 60 * 1000; // 1 minute

export async function executeMerge(params: {
  missionId: string;
  repoPath: string;
  sourceBranch: string;
  targetBranch: string;
  mission: Mission;
  claudeMdPath?: string | null;
  onRetryScheduled?: (retryAt: number) => void;
}): Promise<MergeResult> {
  const { repoPath, sourceBranch, targetBranch, mission, claudeMdPath, onRetryScheduled } = params;

  // Attempt 1
  const result1 = await attemptMerge(repoPath, sourceBranch, targetBranch, mission, claudeMdPath);
  if (result1.success) return result1;

  // Only retry on conflict failures, not other git errors
  if (!result1.error?.includes('onflict')) {
    return result1;
  }

  // Schedule retry
  const retryAt = Date.now() + RETRY_DELAY_MS;
  onRetryScheduled?.(retryAt);

  // Wait 60 seconds for other merges to land
  await new Promise(resolve => setTimeout(resolve, RETRY_DELAY_MS));

  // Fetch latest main before retry
  const git = simpleGit(repoPath);
  try { await git.fetch(); } catch { /* continue with what we have */ }

  // Attempt 2
  return attemptMerge(repoPath, sourceBranch, targetBranch, mission, claudeMdPath);
}

async function attemptMerge(
  repoPath: string,
  sourceBranch: string,
  targetBranch: string,
  mission: Mission,
  claudeMdPath?: string | null,
): Promise<MergeResult> {
  const git = simpleGit(repoPath);

  try {
    let stashed = false;
    const status = await git.status();
    if (status.modified.length > 0 || status.staged.length > 0 || status.not_added.length > 0) {
      await git.stash(['push', '-m', `devroom-pre-merge-${sourceBranch}`]);
      stashed = true;
    }

    await git.checkout(targetBranch);

    try {
      await git.merge([sourceBranch, '--no-ff']);
    } finally {
      if (stashed) {
        try { await git.stash(['pop']); } catch {
          console.warn(`[Quartermaster] Stash pop failed after merging ${sourceBranch}.`);
        }
      }
    }

    return { success: true, conflictResolved: false };
  } catch (err) {
    const errorMsg = err instanceof Error ? err.message : String(err);

    if (errorMsg.includes('CONFLICTS') || errorMsg.includes('conflict')) {
      const resolved = await resolveConflicts({
        repoPath, sourceBranch, targetBranch, mission, claudeMdPath,
      });

      if (resolved) {
        return { success: true, conflictResolved: true };
      }

      try { await git.merge(['--abort']); } catch { /* may already be aborted */ }
      return {
        success: false,
        conflictResolved: false,
        error: 'Conflict resolution failed. Branch preserved for manual review.',
      };
    }

    return { success: false, conflictResolved: false, error: errorMsg };
  }
```

```typescript
/**
 * Triggered by Overseer after approving a mission.
 * Handles merge (if worktree), cleanup, and follow-up extraction.
 */
export async function triggerQuartermaster(missionId: string): Promise<void> {
  const db = getDatabase();

  const mission = db.select().from(missions).where(eq(missions.id, missionId)).get();
  if (!mission || mission.status !== 'approved') {
    console.warn(`[Quartermaster] Skipping ${missionId} — status is ${mission?.status}, not approved`);
    return;
  }

  const battlefield = db.select().from(battlefields)
    .where(eq(battlefields.id, mission.battlefieldId)).get();
  if (!battlefield) {
    console.error(`[Quartermaster] Battlefield not found for mission ${missionId}`);
    return;
  }

  // Non-worktree missions: skip merge, go directly to accomplished
  if (!mission.worktreeBranch) {
    db.update(missions).set({
      status: 'accomplished',
      completedAt: Date.now(),
      updatedAt: Date.now(),
    }).where(eq(missions.id, missionId)).run();

    emitStatusChange('mission', missionId, 'accomplished');
    emitMissionLog(missionId, '[Quartermaster] No worktree — mission accomplished.');
    cleanupMissionHome(missionId);

    await extractSuggestionsAndNotify(mission as Mission, battlefield.id);
    await notifyCampaignIfNeeded(mission as Mission);
    return;
  }

  // Worktree missions: merge flow
  db.update(missions).set({
    status: 'merging',
    updatedAt: Date.now(),
  }).where(eq(missions.id, missionId)).run();

  emitStatusChange('mission', missionId, 'merging');
  emitMissionLog(missionId, `[Quartermaster] Merging ${mission.worktreeBranch} into ${battlefield.defaultBranch || 'main'}...`);

  const worktreePath = path.join(
    battlefield.repoPath, '.worktrees',
    mission.worktreeBranch.replace(/\//g, '-'),
  );

  const mergeResult = await executeMerge({
    missionId: mission.id,
    repoPath: battlefield.repoPath,
    sourceBranch: mission.worktreeBranch,
    targetBranch: battlefield.defaultBranch || 'main',
    mission: mission as Mission,
    claudeMdPath: battlefield.claudeMdPath,
    onRetryScheduled: (retryAt) => {
      db.update(missions).set({ mergeRetryAt: retryAt, updatedAt: Date.now() })
        .where(eq(missions.id, missionId)).run();
      emitStatusChange('mission', missionId, 'merging', { mergeRetryAt: retryAt });
      emitMissionLog(missionId, `[Quartermaster] First merge attempt failed. Retrying in 60s...`);
    },
  });

  // Clear retryAt
  db.update(missions).set({ mergeRetryAt: null }).where(eq(missions.id, missionId)).run();

  if (mergeResult.success) {
    // Clean up worktree
    await removeWorktree(battlefield.repoPath, worktreePath, mission.worktreeBranch);
    cleanupMissionHome(missionId);

    const logMsg = mergeResult.conflictResolved
      ? `[Quartermaster] Merged ${mission.worktreeBranch} (conflicts auto-resolved). Worktree cleaned.`
      : `[Quartermaster] Merged ${mission.worktreeBranch}. Worktree cleaned.`;

    db.update(missions).set({
      status: 'accomplished',
      completedAt: Date.now(),
      updatedAt: Date.now(),
    }).where(eq(missions.id, missionId)).run();

    emitStatusChange('mission', missionId, 'accomplished');
    emitMissionLog(missionId, logMsg);

    await extractSuggestionsAndNotify(mission as Mission, battlefield.id);
    await notifyCampaignIfNeeded(mission as Mission);
  } else {
    // Merge failed — clean up config but preserve branch
    cleanupMissionHome(missionId);

    db.update(missions).set({
      status: 'compromised',
      compromiseReason: 'merge-failed',
      debrief: (mission.debrief || '') + `\n\n---\n\nMERGE FAILED: ${mergeResult.error}\nBranch \`${mission.worktreeBranch}\` preserved for inspection.`,
      completedAt: Date.now(),
      updatedAt: Date.now(),
    }).where(eq(missions.id, missionId)).run();

    emitStatusChange('mission', missionId, 'compromised', { compromiseReason: 'merge-failed' });
    emitMissionLog(missionId, `[Quartermaster] Merge failed: ${mergeResult.error}`);

    await escalate({
      level: 'warning',
      title: `Merge failed: ${mission.title}`,
      detail: `${mergeResult.error}. Branch \`${mission.worktreeBranch}\` preserved.`,
      entityType: 'mission',
      entityId: mission.id,
      battlefieldId: mission.battlefieldId,
    });

    await notifyCampaignIfNeeded(mission as Mission);
  }
}
```

Create `src/lib/utils/dependency-graph.ts`:

```typescript
interface DependencyNode {
  title: string;
  dependsOn: string[];
}

/**
 * Detect cycles in a dependency graph using DFS.
 * Returns a string describing the cycle path, or null if acyclic.
 */
export function detectCycle(nodes: DependencyNode[]): string | null {
  const adjacency = new Map<string, string[]>();
  for (const node of nodes) {
    adjacency.set(node.title, node.dependsOn);
  }

  const visited = new Set<string>();
  const inStack = new Set<string>();

  for (const node of nodes) {
    const cycle = dfs(node.title, adjacency, visited, inStack, []);
    if (cycle) return cycle;
  }

  return null;
}

function dfs(
  node: string,
  adjacency: Map<string, string[]>,
  visited: Set<string>,
  inStack: Set<string>,
  path: string[],
): string | null {
  if (inStack.has(node)) {
    const cycleStart = path.indexOf(node);
    const cyclePath = [...path.slice(cycleStart), node];
    return `Circular dependency: ${cyclePath.join(' -> ')}`;
  }
  if (visited.has(node)) return null;

  visited.add(node);
  inStack.add(node);
  path.push(node);

  for (const dep of adjacency.get(node) ?? []) {
    const cycle = dfs(dep, adjacency, visited, inStack, path);
    if (cycle) return cycle;
  }

  path.pop();
  inStack.delete(node);
  return null;
}
```

```tsx
'use client';

import { useState, useEffect } from 'react';

export function MergeCountdown({ retryAt }: { retryAt: number }) {
  const [remaining, setRemaining] = useState(Math.max(0, Math.ceil((retryAt - Date.now()) / 1000)));

  useEffect(() => {
    const interval = setInterval(() => {
      const r = Math.max(0, Math.ceil((retryAt - Date.now()) / 1000));
      setRemaining(r);
      if (r <= 0) clearInterval(interval);
    }, 1000);
    return () => clearInterval(interval);
  }, [retryAt]);

  if (remaining <= 0) return <span className="text-xs text-tac-amber">Retrying...</span>;
  return <span className="text-xs text-tac-amber">Retry in {remaining}s</span>;
}
```
